Step-by-Step Setup Guide

  1. Create a new Google Sheet and note its ID (from the URL)
  2. Create a new Google Apps Script project
  3. Copy the provided Apps Script code into the project
  4. Deploy the script as a Web App with these settings:
    • Execute as: Me
    • Who has access: Anyone (or "Anyone with Google account")
  5. Copy the Web App URL and paste it above
  6. Click "Connect to Sheets"

Important Notes

  • The first time you access the script URL, you must authorize it in your browser
  • Google may show a warning screen - click "Advanced" then "Go to [your project] (unsafe)"
  • Make sure your script deployment is set to "Anyone" not just "Anyone in your organization"

Connection Troubleshooting

Common "Failed to Fetch" Solutions

  • Authorization Required - Open your script URL directly in a browser and authorize it
  • CORS Issue - Make sure your script handles OPTIONS requests (see updated script below)
  • URL Format - Your URL must end with "/exec" not "/dev"
  • Deployment Settings - Verify "Execute as me" and "Anyone" access
  • Mixed Content - Ensure your HTML page is served over HTTPS

function handleRequest(e) {
  // Handle CORS preflight
  if (e.httpMethod === 'OPTIONS') {
    return handleCors();
  }

  const action = e.parameter.action;
  const sheetId = e.parameter.sheetId;
  
  try {
    let result;
    const sheet = sheetId ? SpreadsheetApp.openById(sheetId) : SpreadsheetApp.getActiveSpreadsheet();
    
    switch (action) {
      case "testConnection":
        result = { success: true, message: "Connection successful" };
        break;
      // ... other cases remain the same ...
      default:
        throw new Error("Invalid action");
    }
    
    return ContentService.createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON)
      .setHeaders({
        'Access-Control-Allow-Origin': '*',
        'Access-Control-Allow-Methods': 'GET, POST, OPTIONS',
        'Access-Control-Allow-Headers': 'Content-Type'
      });
    
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({ 
      error: error.message 
    }))
    .setMimeType(ContentService.MimeType.JSON)
    .setHeaders({
      'Access-Control-Allow-Origin': '*',
      'Access-Control-Allow-Methods': 'GET, POST, OPTIONS',
      'Access-Control-Allow-Headers': 'Content-Type'
    })
    .setStatusCode(500);
  }
}

function handleCors() {
  return ContentService.createTextOutput()
    .setMimeType(ContentService.MimeType.TEXT)
    .setHeaders({
      'Access-Control-Allow-Origin': '*',
      'Access-Control-Allow-Methods': 'POST, GET, OPTIONS',
      'Access-Control-Allow-Headers': 'Content-Type'
    });
}
